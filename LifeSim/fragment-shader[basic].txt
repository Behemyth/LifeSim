#version 420 core


in vec3 vert_FS_in;
in vec3 normal_FS_in;
in vec3 color_FS_in;

out vec4 fragment;

vec4 ambientLight=vec4(0.05,0.05,0.05,1.0); 
vec4 lightColor=vec4(0.9,0.9,0.9,1.0); 

vec3 lightNormal=vec3(0.0f,0.0f,-1.0f);
	

void main(){

	vec3 normal = normalize(normal_FS_in);

	float intensity = max(dot(lightNormal, normal),0.0);


	vec3 calcColor = mix(color_FS_in.rgb*(0.4f),color_FS_in.rgb,intensity);
	fragment = vec4(calcColor, 1.0f);
}
